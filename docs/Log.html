	def print(message, without_dump_file = True):
#

Prints the given message to the console and, if without_dump_file is False, saves the message to a dump file.

:param message: The message to be printed and/or saved.

:type message: str

:param without_dump_file: Whether or not to save the message to a dump file. Default is True.

:type without_dump_file: bool

		now = datetime.now()
		current_time = str(now.strftime("%H:%M:%S"))
		today = str(date.today())
		msg = f"[{current_time}] {message}"
		path = f"logs/dump-vnc-{today}.log"

		if without_dump_file == True:
			print(msg)
			return

		with open(path, 'a') as file:
			file.write(msg + "\n")
			print(msg)
